12  /**
13   * Class to search csv file given item as a string to search for, user can provide column name,
14   * index, or none to search entire file.
15   */
16  public class Search {
17  
18    private final List<String> headerList;
19    private final List<List<String>> listSearchThrough;
20  
21    /**
22     * Constructor to build a Search object to begin searching in target csv file. May search with
23     * given column name, column index number, or no column specification. Create Parse object to
24     * read-in csv file and convert each row to a list of object.
25     *
26     * @param fileName file path under data directory (data/ (fill in))
27     * @param hasHeader true if csv has header, false otherwise
28     * @throws FactoryFailureException when parser fails to convert each row into List of string
29     *     object
30     * @throws IOException when invalid file name or file not found
31     */
32    public Search(String fileName, boolean hasHeader) throws FactoryFailureException, IOException {
33      // to make files only under data directory accessible
34      if (fileName.charAt(0) == '.') {
35        System.out.println(fileName.charAt(0));
36        throw new IOException("invalid file name");
37      }
38      Reader fileReader = new FileReader("data/" + fileName);
39  
40      // create parser and call parse
41      CSVParser<List<String>> parser =
42          new CSVParser<>(fileReader, new StringListCreator(), hasHeader);
43      parser.parse();
44      this.listSearchThrough = parser.getParseResult();
45      this.headerList = parser.getParseHeader();
46    }

48    /**
49     * Compares string a to b after trimming and replacing all whitespaces. Capitalization are viewed
50     * as different strings. (Used in Search to see if item an is the same as item b).
51     *
52     * @param a string a
53     * @param b string b
54     * @return true if string an is equal to string b, false otherwise
55     */
56    public boolean checkStringEqual(String a, String b) {
57      a = a.trim().replaceAll("\\s", "");
58      b = b.trim().replaceAll("\\s", "");
59      return a.equals(b);
60    }

62    /**
63     * Print out the row in given csv that contains the item we're looking for according to column
64     * identification specifications. When row not found, not found messages are printed instead
65     * according to given column identification.
66     *
67     * @param row item found in this row, null if item not found
68     * @param searchFor item to search for
69     * @param columnName name of the column item was found if provided column name, null if no column
70     *     name specified
71     * @param colIndex the index of column item was found if provided column index, -1 if no column
72     *     index specified
73     * @param found true if item was found, false otherwise
74     */
75    public void printFoundRow(
76        List<String> row, String searchFor, String columnName, int colIndex, boolean found) {
77      if (found) {
78        if (columnName != null) {
79          System.out.println(
80              "Row containing "
81                  + searchFor
82                  + " under column"
83                  + columnName
84                  + ": "
85                  + String.join(", ", row));
86        } else if (colIndex >= 0) {
87          System.out.println(
88              "Row containing "
89                  + searchFor
90                  + " under column number "
91                  + colIndex
92                  + ": "
93                  + String.join(", ", row));
94        } else {
95          System.out.println("Row containing " + searchFor + ": " + String.join(", ", row));
96        }
97      } else {
98        if (columnName != null) {
99          System.out.println(searchFor + "not found in given csv under column name " + columnName);
100       } else if (colIndex >= 0) {
101         System.out.println(searchFor + "not found in given csv under column index " + colIndex);
102       } else {
103         System.out.println(searchFor + "not found in given csv");
104       }
105     }
106   }

108   /**
109    * To exclude mismatch rows based on row sizes when searching for item. If header is provided,
110    * standard row size is determined by size of list holding header items. If header not provided,
111    * standard size is determined by the first row in csv. Ensures search functions search withing
112    * index boundary.
113    *
114    * @param listOfRows containing main body csv rows (exclude header if header provided, entire csv
115    *     if no header)
116    * @param headerList if not empty (csv has header), all rows must be header size (set as standard
117    *     size) to be considered as matched row in search
118    * @return the filtered list of main body csv rows with matching size ready for search
119    */
120   public List<List<String>> makeSearchable(List<List<String>> listOfRows, List<String> headerList) {
121     int expectedRowSize;
122     if (!headerList.isEmpty()) {
123       expectedRowSize = headerList.size();
124     } else {
125       expectedRowSize = listOfRows.get(0).size();
126     }
127 
128     List<List<String>> resultList = new ArrayList<>();
129     for (List<String> row : listOfRows) {
130       if (row.size() != expectedRowSize) {
131         System.out.println("mismatched to row format: " + row);
132       } else {
133         resultList.add(row);
134       }
135     }
136     return resultList;
137   }

139   /**
140    * A search function to look for target item in csv given an index for the column to search under.
141    * Print out result to user interface (See printFoundRow() in Search class).
142    *
143    * @param searchFor item to look for
144    * @param columnIndex index of column to search under
145    * @return list holding rows found with item under this provided column given column index
146    * @throws ArrayIndexOutOfBoundsException if index provided is out of bound for csv file
147    */
148   public List<List<String>> startSearch(String searchFor, int columnIndex)
149       throws ArrayIndexOutOfBoundsException {
150     List<List<String>> foundRows = new ArrayList<>();
151     List<List<String>> searchableList =
152         this.makeSearchable(this.listSearchThrough, this.headerList);
153 
154     for (List<String> row : searchableList) {
155       if (this.checkStringEqual(searchFor, row.get(columnIndex))) {
156         this.printFoundRow(row, searchFor, null, columnIndex, true);
157         foundRows.add(row);
158       }
159     }
160 
161     if (foundRows.isEmpty()) {
162       this.printFoundRow(null, searchFor, null, columnIndex, false);
163     }
164 
165     return foundRows;
166   }

168   /**
169    * A search function to look for target item in csv given a name for the column to search. Print
170    * out result to user interface (See printFoundRow() in Search class). Search will only happen if
171    * header row exists and if given column name is found in header list.
172    *
173    * @param searchFor item to search for
174    * @param columnName name of column to search in this column only
175    * @return list holding rows found with item under this provided column name
176    */
177   public List<List<String>> startSearch(String searchFor, String columnName) {
178     List<List<String>> foundRows = new ArrayList<>();
179     List<List<String>> searchableList =
180         this.makeSearchable(this.listSearchThrough, this.headerList);
181 
182     if (!this.headerList.isEmpty()) {
183       int columnIndex = -1;
184       // find index of according column containing provided header name
185       for (int i = 0; i < this.headerList.size(); i++) {
186         if (this.checkStringEqual(this.headerList.get(i).trim(), columnName)) {
187           columnIndex = i;
188         }
189       }
190 
191       // header name found, search with columnIndex
192       if (columnIndex != -1) {
193         for (List<String> row : searchableList) {
194           if (this.checkStringEqual(searchFor, row.get(columnIndex))) {
195             this.printFoundRow(row, searchFor, columnName, -1, true);
196             foundRows.add(row);
197           }
198         }
199         if (foundRows.isEmpty()) {
200           this.printFoundRow(null, searchFor, columnName, -1, false);
201         }
202       } else {
203         System.out.println("Header name not found: " + columnName);
204       }
205     }
206     return foundRows;
207   }

209   /**
210    * A search function to look through each row of csv file for item without column identifications.
211    * Print out result to user interface (See printFoundRow() in Search class).
212    *
213    * @param searchFor item to search for in entire csv file
214    * @return list holding rows found with item
215    */
216   public List<List<String>> startSearch(String searchFor) {
217     List<List<String>> foundRows = new ArrayList<>();
218     List<List<String>> searchableList =
219         this.makeSearchable(this.listSearchThrough, this.headerList);
220     for (List<String> row : searchableList) {
221       for (String item : row) {
222         if (this.checkStringEqual(searchFor, item)) {
223           this.printFoundRow(row, searchFor, null, -1, true);
224           foundRows.add(row);
225         }
226       }
227     }
228     if (foundRows.isEmpty()) {
229       this.printFoundRow(null, searchFor, null, -1, false);
230     }
231     return foundRows;
232   }
